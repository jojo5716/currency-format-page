CurrencyFormat.com Is a tiny JavaScript library, providing simple and advanced money and currency formatting
Demo 

                


                

                    Usage

                    
    
                        
  1. 
                            Download the script and put it in your project, then reference it in your HTML. 
                            
                                    
                                   
    1. <script type="text/javascript" src="path/to/file/js/currency-format.js"></script>
    <script type="text/javascript" src="path/to/file/js/currency-format.js"></script>
                                  
                                        
                            
    
                            
                        
    2. 
                        
  2. 
                            Call method currencies.formatMoney() to convert the price
                            
                                    
                                    
    1. var price = currencies.formatMoney();
    2. // € 0.00    
    4. var price = currencies.formatMoney(2000.65, 'EUR');
    5. // € 2,000.65 
    7. var price = currencies.formatMoney(2000.65, 'EUR', 'es_ES');
    8. // 2.000,65 €
    10. var price = currencies.formatMoney(2000.65, 'EUR', 'es_ES', 3);
    11. // 2.000,650 € 
    13. var price = currencies.formatMoney(2000.65, 'EUR', 'es_ES', 3, ',');
    14. // 2,000,650 €
    16. var price = currencies.formatMoney(2000.65, 'EUR', 'es_ES', 3, ',', '.');
    17. // 2,000.650 €
